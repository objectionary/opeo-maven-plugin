430     /**
431      * Matcher for the same node.
432      * @since 0.2
433      */
434     private static final class SameNode extends TypeSafeMatcher<Iterable<Directive>> {
435 
436         /**
437          * Expected node.
438          */
439         private final AstNode expected;
440 
441         /**
442          * Expected node.
443          */
444         private final AtomicReference<String> exp;
445 
446         /**
447          * Actual node.
448          */
449         private final AtomicReference<String> actual;
450 
451         /**
452          * Constructor.
453          * @param expected Expected node.
454          */
455         private SameNode(final AstNode expected) {
456             this.expected = expected;
457             this.exp = new AtomicReference<>();
458             this.actual = new AtomicReference<>();
459         }
460 
461         @Override
462         public void describeTo(final Description description) {
463             description.appendValue(this.exp.get());
464         }
465 
466         @Override
467         public boolean matchesSafely(final Iterable<Directive> item) {
468             try {
469                 final String xactual = new Xembler(item).xml();
470                 final String xpected = new Xembler(this.expected.toXmir()).xml();
471                 this.actual.set(xactual);
472                 this.exp.set(xpected);
473                 return new SameXml(xpected).matchesSafely(xactual);
474             } catch (final ImpossibleModificationException exception) {
475                 throw new IllegalStateException(
476                     String.format(
477                         "Can't convert to XML the following instructions: %n%s%n",
478                         item
479                     ),
480                     exception
481                 );
482             }
483         }
484 
485         @Override
486         public void describeMismatchSafely(
487             final Iterable<Directive> item, final Description mismatch
488         ) {
489             mismatch.appendText("was ").appendValue(this.actual.get());
490         }
